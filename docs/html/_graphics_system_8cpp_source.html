213  void GraphicsSystem::handleWindowEvent(const SDL_Event& evt) {
214  switch (evt.window.event) {
215  case SDL_WINDOWEVENT_SIZE_CHANGED: int w, h; SDL_GetWindowSize(mSdlWindow, &w, &h);
216 #if OGRE_PLATFORM == OGRE_PLATFORM_LINUX
217  mRenderWindow->resize(w, h);
218 #else
219  mRenderWindow->windowMovedOrResized();
220 #endif
221  break;
222  case SDL_WINDOWEVENT_RESIZED:
223 #if OGRE_PLATFORM == OGRE_PLATFORM_LINUX
224  mRenderWindow->resize(evt.window.data1, evt.window.data2);
225 #else
226  mRenderWindow->windowMovedOrResized();
227 #endif
228  break;
229  case SDL_WINDOWEVENT_CLOSE: break;
230  case SDL_WINDOWEVENT_SHOWN: mRenderWindow->setVisible(true); break;
231  case SDL_WINDOWEVENT_HIDDEN: mRenderWindow->setVisible(false); break;

235  void GraphicsSystem::processIncomingMessage(Mq::MessageId messageId, const void* data) {
236  BaseSystem::processIncomingMessage(messageId, data);
237  switch (messageId) {
238  case Mq::LOGICFRAME_FINISHED: {
239  Ogre::uint32 newIdx = *reinterpret_cast<const Ogre::uint32*>(data);
240 
241  if (newIdx != std::numeric_limits<Ogre::uint32>::max()) {
242  mAccumTimeSinceLastLogicFrame = 0;
243  this->queueSendMessage(
244  mLogicSystem,
245  Mq::LOGICFRAME_FINISHED,
246  (mCurrentTransformIdx + NUM_GAME_ENTITY_BUFFERS - 1) % NUM_GAME_ENTITY_BUFFERS);
247 
248  assert((mCurrentTransformIdx + 1) % NUM_GAME_ENTITY_BUFFERS == newIdx && "Graphics is receiving indices out of order!!!");
249 
250  mCurrentTransformIdx = newIdx;
251  }
252  } break;
253  case Mq::GAME_ENTITY_ADDED:
254  gameEntityAdded(
255  reinterpret_cast<const GameEntityManager::CreatedGameEntity*>(data));
256  break;
257  case Mq::GAME_ENTITY_REMOVED:
258  gameEntityRemoved(*reinterpret_cast<GameEntity* const*>(data));
259  break;
260  case Mq::GAME_ENTITY_SCHEDULED_FOR_REMOVAL_SLOT:
261  this->queueSendMessage(mLogicSystem,
262  Mq::GAME_ENTITY_SCHEDULED_FOR_REMOVAL_SLOT,
263  *reinterpret_cast<const Ogre::uint32*>(data));
264  break;
265  default: break;
266  }
267  }

413  const Ogre::Matrix4* RESTRICT_ALIAS _r) const {
414  const Ogre::Transform& transform = _l->mSceneNode->_getTransform();
415  return &transform.mDerivedTransform[transform.mIndex] < _r;

426  const Ogre::Transform& rTransform = _r->mSceneNode->_getTransform();
427  return &lTransform.mDerivedTransform[lTransform.mIndex] < &rTransform.mDerivedTransform[rTransform.mIndex];
428  }

431  void GraphicsSystem::gameEntityAdded(const GameEntityManager::CreatedGameEntity* cge) {
432  Ogre::SceneNode* sceneNode = mSceneManager->getRootSceneNode(cge->gameEntity->mType)
433  ->createChildSceneNode(cge->gameEntity->mType,
434  cge->initialTransform.vPos,
435  cge->initialTransform.qRot);
436  sceneNode->setScale(cge->initialTransform.vScale);
437  cge->gameEntity->mSceneNode = sceneNode;
438 
439  if (cge->gameEntity->mMoDefinition->moType == MoTypeItem) {
440  Ogre::Item* item = mSceneManager->createItem(
441  cge->gameEntity->mMoDefinition->meshName,
442  cge->gameEntity->mMoDefinition->resourceGroup,
443  cge->gameEntity->mType);
444 
445  Ogre::StringVector materialNames = cge->gameEntity->mMoDefinition->submeshMaterials;
446  size_t minMaterials = std::min(materialNames.size(), item->getNumSubItems());
447 
448  for (size_t i = 0; i < minMaterials; ++i) {
449  item->getSubItem(i)->setDatablockOrMaterialName(
450  materialNames[i],
451  cge->gameEntity->mMoDefinition->resourceGroup);
452  }
453 
454  cge->gameEntity->mMovableObject = item;
455  }
456 
457  sceneNode->attachObject(cge->gameEntity->mMovableObject);
458 
459  // Linking Graphics and Physics
460  if (dynamic_cast<btRigidBody*>(cge->gameEntity->mCollisionObject)) {
461  btTransform transform = Collision::Converter::to(cge->initialTransform);
462  static_cast<btRigidBody*>(cge->gameEntity->mCollisionObject)->setMotionState(new Collision::ObjectState(transform, cge->gameEntity, (LogicSystem*) mLogicSystem));
463  }
464 
465  const Ogre::Transform& transform = sceneNode->_getTransform();
466  GameEntityVec::iterator itGameEntity = std::lower_bound(mGameEntities[cge->gameEntity->mType].begin(),
467  mGameEntities[cge->gameEntity->mType].end(),
468  &transform.mDerivedTransform[transform.mIndex],
469  GameEntityCmp());
470  mGameEntities[cge->gameEntity->mType].insert(itGameEntity, cge->gameEntity);
471  }

500  void GraphicsSystem::execute(size_t threadId, size_t numThreads) {
501  size_t currIdx = mCurrentTransformIdx;
502  size_t prevIdx = (mCurrentTransformIdx + NUM_GAME_ENTITY_BUFFERS - 1) % NUM_GAME_ENTITY_BUFFERS;
503 
504  const size_t objsPerThread = (mThreadGameEntityToUpdate->size() + (numThreads - 1)) / numThreads;
505  const size_t toAdvance = std::min(threadId * objsPerThread, mThreadGameEntityToUpdate->size());
506 
507  GameEntityVec::const_iterator itor = mThreadGameEntityToUpdate->begin() + toAdvance;
508  GameEntityVec::const_iterator end = mThreadGameEntityToUpdate->begin() + std::min(toAdvance + objsPerThread, mThreadGameEntityToUpdate->size());
509  while (itor != end) {
510  GameEntity* gEnt = *itor;
511  Ogre::Vector3 interpVec = Ogre::Math::lerp(gEnt->mTransform[prevIdx]->vPos,
512  gEnt->mTransform[currIdx]->vPos,
513  mThreadWeight);
514  gEnt->mSceneNode->setPosition(interpVec);
515 
516  interpVec = Ogre::Math::lerp(gEnt->mTransform[prevIdx]->vScale,
517  gEnt->mTransform[currIdx]->vScale,
518  mThreadWeight);
519  gEnt->mSceneNode->setScale(interpVec);
520 
521  Ogre::Quaternion interpQ = Ogre::Math::lerp(gEnt->mTransform[prevIdx]->qRot,
522  gEnt->mTransform[currIdx]->qRot,
523  mThreadWeight);
524  gEnt->mSceneNode->setOrientation(interpQ);
525 
526  ++itor;
527  }
528  }
