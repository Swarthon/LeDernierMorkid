28  void CameraController::update(float timeSinceLast) {
29  Ogre::Camera* camera = mGraphicsSystem->getCamera();
30 
31  if (mCameraYaw || mCameraPitch) {
32  if (mUseSceneNode) {
33  Ogre::Node* cameraNode = camera->getParentNode();
34 
35  // Update now as yaw needs the derived orientation.
36  cameraNode->_getFullTransformUpdated();
37  cameraNode->yaw(Ogre::Radian(mCameraYaw), Ogre::Node::TS_WORLD);
38  cameraNode->pitch(Ogre::Radian(mCameraPitch));
39  }
40  else {
41  camera->yaw(Ogre::Radian(mCameraYaw));
42  camera->pitch(Ogre::Radian(mCameraPitch));
43  }
44 
45  mCameraYaw = 0.0f;
46  mCameraPitch = 0.0f;
47  }
48 
49  int camMovementZ = mKeymapState[Backward].second - mKeymapState[Forward].second;
50  int camMovementX = mKeymapState[Rightward].second - mKeymapState[Leftward].second;
51  int slideUpDown = mKeymapState[Up].second - mKeymapState[Down].second;
52 
53  Ogre::Vector3 camMovementDir(camMovementX, slideUpDown, camMovementZ);
54  camMovementDir.normalise();
55  camMovementDir *= timeSinceLast * mCameraBaseSpeed * (1 + mSpeedModifier * mCameraSpeedBoost);
56  if (camMovementZ || camMovementX || slideUpDown) {
57  if (mUseSceneNode) {
58  Ogre::Node* cameraNode = camera->getParentNode();
59  cameraNode->translate(camMovementDir, Ogre::Node::TS_LOCAL);
60  }
61  camera->moveRelative(camMovementDir);
62  }
63  }
