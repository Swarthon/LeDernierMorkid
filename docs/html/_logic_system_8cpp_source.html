58  void LogicSystem::finishFrameParallel(void) {
59  if (mGameEntityManager)
60  mGameEntityManager->finishFrameParallel();
61 
62  // Notify the GraphicsSystem we're done rendering this frame.
63  if (mGraphicsSystem) {
64  size_t idxToSend = mCurrentTransformIdx;
65 
66  if (mAvailableTransformIdx.empty()) {
67  // Don't relinquish our only ID left.
68  // If you end up here too often, Graphics' thread is too slow,
69  // or you need to increase NUM_GAME_ENTITY_BUFFERS
70  idxToSend = std::numeric_limits<Ogre::uint32>::max();
71  }
72  else {
73  // Until Graphics constantly releases the indices we send them, to
74  // avoid writing
75  // to transform data that may be in use by the other thread (race
76  // condition)
77  mCurrentTransformIdx = mAvailableTransformIdx.front();
78  mAvailableTransformIdx.pop_front();
79  }
80 
81  this->queueSendMessage(mGraphicsSystem, Mq::LOGICFRAME_FINISHED, idxToSend);
82  }
83 
84  BaseSystem::finishFrameParallel();
85  }

87  void LogicSystem::processIncomingMessage(Mq::MessageId messageId, const void* data) {
88  BaseSystem::processIncomingMessage(messageId, data);
89  switch (messageId) {
90  case Mq::LOGICFRAME_FINISHED: {
91  Ogre::uint32 newIdx = *reinterpret_cast<const Ogre::uint32*>(data);
92  assert((mAvailableTransformIdx.empty() || newIdx == (mAvailableTransformIdx.back() + 1) % NUM_GAME_ENTITY_BUFFERS) && "Indices are arriving out of order!!!");
93 
94  mAvailableTransformIdx.push_back(newIdx);
95  } break;
96  case Mq::GAME_ENTITY_SCHEDULED_FOR_REMOVAL_SLOT:
97  mGameEntityManager->_notifyGameEntitiesRemoved(
98  *reinterpret_cast<const Ogre::uint32*>(data));
99  break;
100  case Mq::SDL_EVENT: break;
101  default: break;
102  }
