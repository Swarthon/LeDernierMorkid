53  void SdlInputHandler::_handleSdlEvents(const SDL_Event& evt) {
54  switch (evt.type) {
55  case SDL_MOUSEMOTION:
56  // Ignore this if it happened due to a warp
57  if (!handleWarpMotion(evt.motion)) {
58  // Try to keep the mouse inside the window
59  if (mWindowHasFocus)
60  wrapMousePointer(evt.motion);
61 
62  emit(evt);
63  }
64  break;
65  case SDL_MOUSEWHEEL:
66  case SDL_MOUSEBUTTONDOWN:
67  case SDL_MOUSEBUTTONUP:
68  case SDL_KEYDOWN:
69  case SDL_KEYUP:
70  case SDL_TEXTINPUT:
71  case SDL_JOYAXISMOTION:
72  case SDL_JOYBUTTONDOWN:
73  case SDL_JOYBUTTONUP:
74  emit(evt);
75  break;
76  case SDL_JOYDEVICEADDED: break;
77  case SDL_JOYDEVICEREMOVED: break;
78  case SDL_WINDOWEVENT: handleWindowEvent(evt); break;
79  }
80  }

97  void SdlInputHandler::updateMouseSettings(void) {
98  mGrabPointer = mWantMouseGrab && mMouseInWindow && mWindowHasFocus;
99  SDL_SetWindowGrab(mSdlWindow, mGrabPointer ? SDL_TRUE : SDL_FALSE);
100 
101  SDL_ShowCursor(mWantMouseVisible || !mWindowHasFocus);
102 
103  bool relative = mWantRelative && mMouseInWindow && mWindowHasFocus;
104  if (mIsMouseRelative == relative)
105  return;
106 
107  mIsMouseRelative = relative;
108 
109  mWrapPointerManually = false;
110 
111  int success = SDL_SetRelativeMouseMode(relative ? SDL_TRUE : SDL_FALSE);
112  if (!relative || (relative && success != 0))
113  mWrapPointerManually = true;
114 
115  SDL_PumpEvents();
116  SDL_FlushEvent(SDL_MOUSEMOTION);
117  }

126  void SdlInputHandler::wrapMousePointer(const SDL_MouseMotionEvent& evt) {
127  if (mIsMouseRelative || !mWrapPointerManually || !mGrabPointer)
128  return;
129 
130  int width = 0;
131  int height = 0;
132 
133  SDL_GetWindowSize(mSdlWindow, &width, &height);
134 
135  const int FUDGE_FACTOR_X = width;
136  const int FUDGE_FACTOR_Y = height;
137 
138  if (evt.x - FUDGE_FACTOR_X < 0 || evt.x + FUDGE_FACTOR_X > width || evt.y - FUDGE_FACTOR_Y < 0 || evt.y + FUDGE_FACTOR_Y > height) {
139  warpMouse(width / 2, height / 2);
140  }
141  }

143  bool SdlInputHandler::handleWarpMotion(const SDL_MouseMotionEvent& evt) {
144  if (!mWarpCompensate)
145  return false;
146 
147  if (evt.x == mWarpX && evt.y == mWarpY) {
148  mWarpCompensate = false;
149  return true;
150  }
151 
152  return false;
153  }
