43 void TerrainCell::setOrigin(const GridPoint& gridPos, Ogre::uint32 horizontalPixelDim, Ogre::uint32 verticalPixelDim, Ogre::uint32 lodLevel) {
44  mGridX = gridPos.x;
45  mGridZ = gridPos.z;
46  mLodLevel = lodLevel;
47 
48  horizontalPixelDim = std::min(horizontalPixelDim, mParentTerrain->mWidth - mGridX);
49  verticalPixelDim = std::min(verticalPixelDim, mParentTerrain->mDepth - mGridZ);
50 
51  mSizeX = horizontalPixelDim;
52  mSizeZ = verticalPixelDim;
53 
54  //Divide by 2^lodLevel and round up
55  horizontalPixelDim = (horizontalPixelDim + (1u << lodLevel) - 1u) >> lodLevel;
56  verticalPixelDim = (verticalPixelDim + (1u << lodLevel) - 1u) >> lodLevel;
57 
58  //Add an extra vertex to fill the gaps to the next TerrainCell.
59  horizontalPixelDim += 1u;
60  verticalPixelDim += 1u;
61 
62  horizontalPixelDim = std::max(horizontalPixelDim, 2u);
63  verticalPixelDim = std::max(verticalPixelDim, 2u);
64 
65  if (mUseSkirts) {
66  //Add two extra vertices & two extra rows for the skirts.
67  horizontalPixelDim += 2u;
68  verticalPixelDim += 2u;
69  }
70 
71  mVerticesPerLine = horizontalPixelDim * 2u + 2u;
72 
73  assert(mVerticesPerLine * (verticalPixelDim - 1u) > 0u);
74 
75  Ogre::VertexArrayObject* vao = mVaoPerLod[Ogre::VpNormal][0];
76  vao->setPrimitiveRange(0, mVerticesPerLine * (verticalPixelDim - 1u));
77 }

79 bool TerrainCell::merge(TerrainCell* next) {
80  bool merged = false;
81 
82  if (mLodLevel == next->mLodLevel) {
83  GridPoint pos;
84  pos.x = mGridX;
85  pos.z = mGridZ;
86  Ogre::uint32 horizontalPixelDim = mSizeX;
87  Ogre::uint32 verticalPixelDim = mSizeZ;
88 
89  if ((this->mGridX + this->mSizeX == next->mGridX ||
90  next->mGridX + next->mSizeX == this->mGridX) &&
91  mGridZ == next->mGridZ && mSizeZ == next->mSizeZ) {
92  //Merge horizontally
93  pos.x = std::min(mGridX, next->mGridX);
94  horizontalPixelDim += next->mSizeX;
95 
96  this->setOrigin(pos, horizontalPixelDim, verticalPixelDim, mLodLevel);
97  merged = true;
98  }
99  else if ((this->mGridZ + this->mSizeZ == next->mGridZ ||
100  next->mGridZ + next->mSizeZ == this->mGridZ) &&
101  mGridX == next->mGridX && mSizeX == next->mSizeX) {
102  //Merge vertically
103  pos.z = std::min(mGridZ, next->mGridZ);
104  verticalPixelDim += next->mSizeZ;
105 
106  this->setOrigin(pos, horizontalPixelDim, verticalPixelDim, mLodLevel);
107  merged = true;
108  }
109  }
110 
111  return merged;
112 }
